> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Working WFS service: CQL + OGC Filter XML → parameterized PostGIS SQL (golden corpus), GetCapabilities/DescribeFeatureType/GetFeature for WFS 1.0/1.1/2.0 with paging/sort/hits/BBOX/propertyName, streaming GeoJSON/GML2/GML3.2/CSV output, WFS-T, GetFeatureById, and auth CQL-Read enforcement.

Gateway already proxies `/geoserver/.../wfs` → `wfs:8080/wfs` with context headers.

**Tech Stack:** Go 1.26, pgx/v5. No parser generator — hand-rolled lexer + recursive descent (fast, zero deps).

```go
type Expr interface{ isExpr() }
type Property struct{ Name string }
type Literal struct{ Value any } // string, float64, bool
type Compare struct{ Op string; Left, Right Expr }   // = <> < <= > >=
type Logic struct{ Op string; Exprs []Expr }          // AND OR
type Not struct{ Expr Expr }
type Like struct{ Prop Expr; Pattern string; CaseInsensitive bool; Negate bool }
type In struct{ Prop Expr; Values []Expr; Negate bool }
type Between struct{ Prop, Lo, Hi Expr; Negate bool }
type IsNull struct{ Prop Expr; Negate bool }
type BBox struct{ Prop string; MinX, MinY, MaxX, MaxY float64; SRS string }
type Spatial struct{ Op string; Prop string; WKT string }   // INTERSECTS WITHIN CONTAINS DISJOINT TOUCHES CROSSES OVERLAPS EQUALS
type DWithin struct{ Prop string; WKT string; Distance float64; Units string; Beyond bool }
type IncludeAll struct{} // INCLUDE keyword

func ParseCQL(s string) (Expr, error)
```

CQL grammar (ECQL subset):
```
expr    := or
or      := and ( OR and )*
and     := unary ( AND unary )*
unary   := NOT unary | primary
primary := '(' expr ')' | INCLUDE | predicate
predicate := prop (('='|'<>'|'<'|'<='|'>'|'>=') value
           | NOT? LIKE string | NOT? ILIKE string
           | NOT? IN '(' value (',' value)* ')'
           | NOT? BETWEEN value AND value
           | IS NOT? NULL)
           | spatialfn
spatialfn := BBOX '(' prop ',' num ',' num ',' num ',' num [',' string] ')'
           | (INTERSECTS|WITHIN|CONTAINS|DISJOINT|TOUCHES|CROSSES|OVERLAPS|EQUALS) '(' prop ',' wkt ')'
           | DWITHIN '(' prop ',' wkt ',' num ',' ident ')'
value   := string | number | TRUE | FALSE | prop
wkt     := everything from geometry keyword (POINT|LINESTRING|POLYGON|MULTI*|GEOMETRYCOLLECTION) through balanced parens
```
Keywords case-insensitive. Strings single-quoted, `''` escape. Property names bare identifiers (dots allowed for ns:prop → strip prefix before colon).

```go
func ParseCQL(s string) (Expr, error) {
	p := &cqlParser{lex: newLexer(s)}
	if err := p.next(); err != nil {
		return nil, err
	}
	e, err := p.parseOr()
	if err != nil {
		return nil, err
	}
	if p.tok.kind != tokEOF {
		return nil, fmt.Errorf("unexpected %q at end of filter", p.tok.text)
	}
	return e, nil
}
```

```go
// ToSQL compiles an Expr to a parameterized WHERE fragment for PostGIS.
// startArg is the first placeholder index (e.g. 1 -> $1).
// geomCol maps property names used in spatial ops (empty = use as-is).
func ToSQL(e Expr, startArg int) (sql string, args []any, err error)
```

Rules: values → `$n` args; property names validated `^[A-Za-z_][A-Za-z0-9_]*$` (error otherwise) and emitted double-quoted; `Like` → `LIKE`/`ILIKE`; `BBox` → `"geom" && ST_MakeEnvelope($1,$2,$3,$4)`; `Spatial` → `ST_Intersects("geom", ST_GeomFromText($1))` etc (Op title-cased into `ST_<Op>`; EQUALS → ST_Equals, DISJOINT → ST_Disjoint...); `DWithin` → `ST_DWithin("geom", ST_GeomFromText($1), $2)` (Beyond → `NOT ST_DWithin(...)`); `IncludeAll` → `TRUE`.

```go
package filter

import (
	"encoding/json"
	"os"
	"path/filepath"
	"reflect"
	"runtime"
	"testing"
)

type corpusCase struct {
	CQL  string `json:"cql"`
	SQL  string `json:"sql"`
	Args []any  `json:"args"`
}

func TestGoldenCorpus(t *testing.T) {
	_, thisFile, _, _ := runtime.Caller(0)
	path := filepath.Join(filepath.Dir(thisFile), "..", "..", "..", "tests", "filter-corpus", "corpus.json")
	raw, err := os.ReadFile(path)
	if err != nil {
		t.Fatal(err)
	}
	var cases []corpusCase
	if err := json.Unmarshal(raw, &cases); err != nil {
		t.Fatal(err)
	}
	if len(cases) < 30 {
		t.Fatalf("corpus too small: %d", len(cases))
	}
	for _, c := range cases {
		e, err := ParseCQL(c.CQL)
		if err != nil {
			t.Errorf("%s: parse: %v", c.CQL, err)
			continue
		}
		sql, args, err := ToSQL(e, 1)
		if err != nil {
			t.Errorf("%s: tosql: %v", c.CQL, err)
			continue
		}
		if sql != c.SQL {
			t.Errorf("%s:\n got  %s\n want %s", c.CQL, sql, c.SQL)
		}
		if len(args) != len(c.Args) {
			t.Errorf("%s: args %v want %v", c.CQL, args, c.Args)
			continue
		}
		for i := range args {
			// JSON numbers decode to float64; normalize both sides
			if !reflect.DeepEqual(args[i], c.Args[i]) {
				t.Errorf("%s: arg[%d] %#v want %#v", c.CQL, i, args[i], c.Args[i])
			}
		}
	}
}
```
